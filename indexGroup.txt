SELECT COUNT(id), YEAR(enrolment_date)
FROM `students` 
GROUP BY YEAR(enrolment_date)

SELECT COUNT(id), office_address
FROM `teachers`
GROUP BY office_address

SELECT exam_id, SUM(vote) / COUNT(exam_id) 
FROM `exam_student` 
GROUP BY exam_id

SELECT exam_id, AVG(vote) 
FROM `exam_student` 
GROUP BY exam_id
